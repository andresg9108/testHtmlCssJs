«While» and «Do While»

    var i = 0
    while(i <= 3){
      console.log(i)
      i++
    }

    i = 0
    do{
      console.log(i)
      i++
    }while(i <= 3)

For

    for (var i = 0; i < 5; i++) {
      console.log(i)
    }

    var miArreglo = [2, 4, 6, 8]

    for (var i = 0; i < miArreglo.length; i++) {
      console.log(miArreglo[i])
    }
