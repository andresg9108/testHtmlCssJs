Operadores lógicos

Operador de igualdad

    console.log(5 == 5) // true
    console.log(5 == 6) // false
    console.log("Hola" == "Hola") // true
    console.log("Hola" == "hola") // false
    console.log("Hola" == "JavaScript") // false
    console.log([1, 2, 3] == [1, 2, 3]) // false: Compara a los arreglos en memoria

    console.log(9 == 9) // true
    console.log(9 == "9") // true
    console.log(9 === "9") // false: Igualdad estricta

Operador de desigualdad

    console.log(9 != 6) // true
    console.log(9 != 9) // false
    console.log("JavaScript" != "JavaScript") // false
    console.log("Hola" != "JavaScript") // true
    console.log([1, 2, 3] != [1, 2, 3]) // true: Compara a los arreglos en memoria

    console.log(9 != 9) // false
    console.log(9 != "9") // false
    console.log(9 !== "9") // true: Desigualdad estricta

Operaores relativos

    console.log(6 > 5) // true
    console.log(3 > 10) // false
    console.log("B" > "A") // true
    console.log("A" > "B") // false
    console.log("ACB" > "ABC") // true
    console.log("AB" > "A") // true

    console.log(5 > 5) // false
    console.log(5 >= 5) // true

Operadores lógicos AND, OR y NOT

    console.log(true && true)
    console.log(true && false)
    console.log(false && true)
    console.log(false && false)

    console.log(true || true)
    console.log(true || false)
    console.log(false || true)
    console.log(false || false)

    console.log(!true)
    console.log(!false)
