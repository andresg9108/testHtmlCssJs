Nota: Los arreglos son mutables. Lo que quiere decir que puedo hacer algo como «arregloNumeros[1] = 125».

Métodos en arreglos

    const miArreglo = [30, 45, 67, 25, 25.5]

    var elemento = miArreglo.shift() // Eliminar primero y retornarlo
    elemento = miArreglo.pop() // Eliminar último y retornarlo
    miArreglo.unshift(139) // Agregar principio
    miArreglo.push(68) // Agregar al final

    // Unir dos arreglos
    const arreglo1 = [1, 2, 3, 4]
    const arreglo2 = [5, 6, 7]
    const arregloFinal = arreglo1.concat(arreglo2)

    // Aplanar un arreglo
    const miArregloMultidimensional = [[1, 2, 3], [4, 5, 6], [7, 8, 9]]
    var resultado = miArregloMultidimensional.flat()

    // Sumar los elementos de un arreglo
    const miArreglo = [1, 2, 3, 4, 5]
    const resultado = miArreglo.reduce((acumulador, dato) => acumulador + dato, 0)
